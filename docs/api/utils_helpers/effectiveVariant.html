String effectiveVariant(String selected, String? tempoDigits) {
  if (selected.toLowerCase() == 'tempo' &&
      tempoDigits != null &&
      tempoDigits.trim().isNotEmpty) {
    return 'Tempo ${tempoDigits.trim()}';
  }
  return selected;
}
